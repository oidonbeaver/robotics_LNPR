110  def motion_update(self, nu, omega, time, noise_rate_pdf):
111  ns = noise_rate_pdf.rvs()
112  pnu = nu + ns[0]*math.sqrt(abs(nu)/time) + ns[1]*math.sqrt(abs(omega)/time)
113  pomega = omega + ns[2]*math.sqrt(abs(nu)/time) + ns[3]*math.sqrt(abs(omega)/time)
114  self.pose = IdealRobot.state_transition(pnu, pomega, time, self.pose)

116 class Mcl:
