84  def one_step(self, time_interval):

86  obs = self.sensor.data(self.pose) if self.sensor else None#センサ値の雑音が入る
87  nu, omega = self.agent.decision(obs)#ここでワンステップごとにエージェントの命令が復活するのでスタックしても止まったままにならない
88  nu, omega = self.bias(nu,omega)#バイアスはエージェントの制御指令に対してかかる
