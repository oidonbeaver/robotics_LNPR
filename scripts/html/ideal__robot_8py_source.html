102  @classmethod
103  def state_transition(cls, nu,omega, time, pose):
104  t0 = pose[2]
105  if math.fabs(omega) < 1e-10:
106  return pose + np.array([nu*math.cos(t0),
107  nu*math.sin(t0),
108  omega])*time
109  else:
110  return pose + np.array([nu/omega*(math.sin(t0+omega*time)-math.sin(t0)),
111  nu/omega*(-math.cos(t0+omega*time)+math.cos(t0)),
112  omega*time])

182  #シミュレーションするときには真値があったほうがいいので、地図座標から観測データを作る
183  @classmethod
184  def observation_function(cls,cam_pose, obj_pos):
185  diff = obj_pos - cam_pose[0:2]
186  phi = math.atan2(diff[1],diff[0]) - cam_pose[2]
187  while phi >= np.pi:
188  phi -= 2*np.pi
189  while phi < -np.pi:
190  phi += 2*np.pi
191  return np.array([np.hypot(*diff),phi]).T
